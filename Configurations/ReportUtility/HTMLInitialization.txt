<script type="text/javascript">

function toggle(id, val) {
    if (val == "open") {
        document.getElementById('subItem' + id).style.display = "";
        document.getElementById('main' + id).innerHTML =
        '&nbsp;<a href="javascript:void(0)" onclick="toggle(' + id + ', \'close\')" class="style7">-</a>&nbsp;';
    } else {
        document.getElementById('subItem' + id).style.display = "none";
        document.getElementById('main' + id).innerHTML =
        '&nbsp;<a href="javascript:void(0)" onclick="toggle(' + id + ', \'open\')" class="style7">+</a>&nbsp;';
    }
}

function testDataSwitch(id, val) {
    if (val == "open") {
        document.getElementById('dataItem' + id).style.display = "";
        document.getElementById('testData' + id).innerHTML =
        '&nbsp;<a href="javascript:void(0)" onclick="testDataSwitch(' + id + ', \'close\')" class="style7">Hide Printable Test Data</a>&nbsp;';
    } else {
        document.getElementById('dataItem' + id).style.display = "none";
        document.getElementById('testData' + id).innerHTML =
        '&nbsp;<a href="javascript:void(0)" onclick="testDataSwitch(' + id + ', \'open\')" class="style7">Show Printable Test Data</a>&nbsp;';
    }
}
